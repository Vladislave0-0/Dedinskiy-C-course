func sum(v, z)
{
    return v + z; 
}

func some_fnc(x)
{
    while(x < 5)
    {
        printf("Hello_world!");
        x = x + 1;
    }

    return 0;
}

func main()
{
    var y = 1;
    while(y < 5)
    {
        printf("Hello_world!");
        y = y + 1;
    }

    sum(1, 2);
    some_fnc(-5);
    if(y == 1)
    {
        printf(123);
    }
    elif(y > 1)
    {
        printf(456);
    }
    else
    {
        printf(789);
    }
    
    return 0;
}
